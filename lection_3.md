## указатели, массивы и строки
& - выдает адрес объекта
* - оператор косвенного доступа, то есть обратный к &: зная адрес, по которому что-то хранится, он может выдать это что-то
int x = 11;
int y = 2;
int z[10];
int* ip;   //ip - указатель на int

ip = &x;  // теперь ip указывает на адрес, по которому хранится x
y = *ip; // y получил значение x через ссылку, которая хранится в указателе ip
*ip = 0; // x теперь равен 0
ip = &z[0];  ip теперь указывает на z[0]
